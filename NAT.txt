NAT

STEP1 : Specify INSIDE and OUTSIDE
!@UTM-PH
conf t
	int g1
		ip nat outside
	int g3
		ip nat inside
	end

STEP2: SPECIFY TRAFFIC

!@UTM-PH
conf t
	ip access-list extended NAT
		permit ip 11.11.11.96 0.0.0.31 any
		end
	
STEP3: CONFIGURE NAT/PAT (1:1)

!@UTM-PH
conf t
	ip nat inside source static 11.11.11.101 208.8.8.100
	!
	ip route 0.0.0.0 0.0.0.0 208.8.8.2


STEP3: 	CONFIGURE NAT/PAT (1:1) with NAT POOL
!@UTM-PH
config t
	ip nat pool NATPOOL 208.8.8.100 208.8.8.199 netmask 255.255.255.0
	ip nat inside source list NAT pool NATPOOL
	end

STEP3:  CONFIGURE PAT (MANY:1) with NAT POOL
!@UTM-ph
clear ip nat translation *
conf t
	no ip nat inside source static 11.11.11.101 208.8.8.100
	ip nat inside source list NAT interface g1 overload
	end
	

sh run | s nat

ip nat inside source list NAT interface g1 overload
